    \quad \textbf{Простое число} --- число которое делиться только на себя и на единицу. Единица и ноль не являются простыми числами.
    \begin{cpp}
int is_prime(int n){
    for(int i = 2; i*i <= n; ++i){
        if(n%i == 0)){
            return false;
        }
    }
    return true;
}
    \end{cpp}

vector<bool> sieve(n, 1);
sieve[0] = sieve[1] = 0;
for(int i = 2; i < n; ++i){
    if(sieve[i]){
        for(int j = i*i; j < n; j+=i){
            sieve[j] = 0;
        }
    }
}
